\begin{smodule}{class-extensionality.ftl}
  \begin{forthel}
    \begin{lemma}[Class Extensionality]
      Let $X, Y$ be classes.
      Assume that every element of $X$ is an element of $Y$ and every element of
      $Y$ is an element of $X$.
      Then $X = Y$.
    \end{lemma}
  \end{forthel}
