  \symdef{singleton}[args=1]{\comp\{#1\comp\}}

  \begin{fdefinition*}[label=1160414603771904]
    Let $a$ be an object.
    The singleton class of $a$ is
    \[\fclass{x}{x = a}.\]
  \end{fdefinition*}

  \begin{fconvention*}
    Let $\singleton{a}$ stand for the singleton class of $a$.
  \end{fconvention*}

  \begin{fdefinition*}[label=6786618161627136]
    A singleton class is a class $A$ such that $A = \singleton{a}$ for some object $a$.
  \end{fdefinition*}
