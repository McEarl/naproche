  \begin{definition}[label=2159753924968448]
    Let $A, B$ be classes.
    The union of $A$ and $B$ is
    \[\fclass{x}{\text{$x \in A$ or $x \in B$}}.\]
  \end{definition}

  \begin{fconvention*}
    Let $\union{A,B}$ stand for the union of $A$ and $B$.
  \end{fconvention*}
