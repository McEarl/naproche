\begin{smodule}[creators={Marcel Sch√ºtz}]{subclass.ftl}
  \importmodule[naproche/examples/preliminaries]{vocabulary.ftl}
  \importmodule[naproche/examples/meta-inf]{common-notation.ftl}

  \symdef{subclass}{\subset}
  \notation{subclass}[eq]{\subseteq}
  \symdef{supclass}{\supset}
  \notation{supclass}[eq]{\supseteq}
  \symdef{propsubclass}{\subsetneq}
  \symdef{propsupclass}{\supsetneq}

  \begin{fdefinition*}[label=3275578358628352,synonyms={subclass/-es,superclass/-es,include/-s}]
    Let $A$ be a class.
    A subclass of $A$ is a class $B$ such that every element of $B$ is an element of $A$.
  \end{fdefinition*}

  \begin{fconvention*}
    Let $B \subclass[eq] A$ stand for $B$ is a subclass of $A$.
    Let $B \subclass A$ stand for $B \subclass[eq] A$.
  \end{fconvention*}

  \begin{fconvention*}
    Let a superclass of $B$ stand for a class $A$ such that $B \subclass[eq] A$.
    Let $B \supclass[eq] A$ stand for $B$ is a superclass of $A$.
    Let $B \supclass A$ stand for $B \subclass[eq] A$.
  \end{fconvention*}

  \begin{fconvention*}
    Let a proper subclass of $A$ stand for a subclass $B$ of $A$ such that $B \neq A$.
    Let $B \propsubclass A$ stand for $B$ is a proper subclass of $A$.
  \end{fconvention*}

  \begin{fconvention*}
    Let a proper superclass of $B$ stand for a superclass $A$ of $B$ such that $A \neq B$.
    Let $B \propsupclass A$ stand for $B$ is a proper superclass of $A$.
  \end{fconvention*}

  \begin{fconvention*}
    Let $A$ includes $B$ stand for $B \subclass[eq] A$.
    Let $B$ is included in $A$ stand for $B \subclass[eq] A$.
  \end{fconvention*}
