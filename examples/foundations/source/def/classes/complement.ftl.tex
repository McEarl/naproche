  \symdef{complement}[args=2]{#1\mathop{\comp\setminus}#2}

  \begin{fdefinition*}[label=7620345041256448]
    Let $A, B$ be classes.
    The complement of $B$ in $A$ is
    \[\fclass{x}{\text{$x \in A$ and $x \notin B$}}.\]
  \end{fdefinition*}

  \begin{fconvention*}
    Let $\complement{A}{B}$ stand for the complement of $B$ in $A$.
  \end{fconvention*}
