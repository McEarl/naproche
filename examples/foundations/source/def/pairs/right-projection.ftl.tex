  \begin{fdefinition*}[label=7579452923969536,synonyms={entry/entries,component/-s}]
    Let $p$ be a pair.
    $\rproj{p}$ is the object $b$ such that $p = \tuple{a,b}$ for some object $a$.

    Let the second entry of $p$ stand for $\rproj{p}$.
    Let the right entry of $p$ stand for $\rproj{p}$.
    Let the second component of $p$ stand for $\rproj{p}$.
    Let the right component of $p$ stand for $\rproj{p}$.
  \end{fdefinition*}
