  \begin{fdefinition*}[label=7786352183607296,synonyms={entry/entries,component/-s}]
    Let $p$ be a pair.
    $\lproj{p}$ is the object $a$ such that $p = \tuple{a,b}$ for some object $b$.

    Let the first entry of $p$ stand for $\lproj{p}$.
    Let the left entry of $p$ stand for $\lproj{p}$.
    Let the first component of $p$ stand for $\lproj{p}$.
    Let the left component of $p$ stand for $\lproj{p}$.
  \end{fdefinition*}
