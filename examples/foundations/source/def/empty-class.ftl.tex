  \importmodule[naproche/examples/preliminaries]{macros.ftl}

  \begin{fproposition*}[label=6252477624090624]
    Let $A$ be a class.
    $A$ is empty iff $A$ has no elements.
  \end{fproposition*}

  \begin{fconvention*}
    Let $A$ is nonempty stand for $A$ is not empty.
  \end{fconvention*}

  \begin{fproposition*}[label=7939928493129728]
    $\emptyset = \compterm{x}{x \neq x}$.
  \end{dproposition*}
