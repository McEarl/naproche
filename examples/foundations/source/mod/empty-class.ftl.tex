    \begin{definition}\printlabel{6252477624090624}
      Let $A$ be a class.
      $A$ is empty iff $A$ has no elements.
    \end{definition}

    Let $A$ is nonempty stand for $A$ is not empty.
  \end{forthel}

  \begin{forthel}
    \begin{definition}\printlabel{7939928493129728}
      $\emptyset = \class{x}{x \neq x}$.
    \end{definition}
  \end{forthel}
