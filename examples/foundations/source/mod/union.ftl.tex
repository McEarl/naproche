    \begin{definition}\printlabel{2159753924968448}
      Let $A, B$ be classes.
      The union of $A$ and $B$ is
      \[\class{x}{\text{$x \in A$ or $x \in B$}}.\]
    \end{definition}

    Let $A \cup B$ stand for the union of $A$ and $B$.
  \end{forthel}
