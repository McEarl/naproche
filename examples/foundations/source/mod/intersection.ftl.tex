    \begin{definition}\printlabel{5744033011859456}
      Let $A, B$ be classes.
      The intersection of $A$ and $B$ is
      \[\class{x}{\text{$x \in A$ and $x \in B$}}.\]
    \end{definition}

    Let $A \cap B$ stand for the intersection of $A$ and $B$.
