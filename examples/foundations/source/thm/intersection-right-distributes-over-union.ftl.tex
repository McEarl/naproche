  \begin{fproposition*}[label=3761728443645952]
    Let $A, B, C$ be classes.
    Then \[\intersection{\union{A,B},C} = \union{\intersection{A,C},\intersection{B,C}}.\]
  \end{fproposition*}
  \begin{fproof}
    $\intersection{\union{A,B},C}
      = \intersection{C,\union{A,B}}
      = \union{\intersection{C,A},\intersection{C,B}}
      = \union{\intersection{A,C},\intersection{B,C}}$.
  \end{fproof}
