  \begin{fproposition*}[label=6931785090859008]
    Let $A$ be a class.
    $\emptyclass \subclass[eq] A$.
  \end{fproposition*}
  \begin{fproof}
    $\emptyclass$ has no elements.
    Hence every element of $\emptyclass$ is contained in $A$.
  \end{fproof}
