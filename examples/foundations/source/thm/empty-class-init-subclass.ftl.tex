  \begin{fproposition*}[label=6931785090859008]
    Let $A$ be a class.
    $\emptyset \subseteq A$.
  \end{fproposition*}
  \begin{fproof*}
    $\emptyset$ has no elements.
    Hence every element of $\emptyset$ is contained in $A$.
  \end{fproof*}
