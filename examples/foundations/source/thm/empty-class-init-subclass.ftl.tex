  \begin{forthel}
    \begin{proposition}\printlabel{6931785090859008}
      Let $A$ be a class.
      $\emptyset \subseteq A$.
    \end{proposition}
    \begin{proof}
      $\emptyset$ has no elements.
      Hence every element of $\emptyset$ is contained in $A$.
    \end{proof}
  \end{forthel}
