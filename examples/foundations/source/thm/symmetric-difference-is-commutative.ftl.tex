  \begin{fproposition*}[label=4363570288525312]
    Let $A, B$ be classes.
    Then \[\symdiff{A}{B} = \symdiff{B}{A}.\]
  \end{fproposition*}
  \begin{fproof}
    $\symdiff{A}{B}
      = \complement{\union{A,B})}{\intersection{A,B}}
      = \complement{\union{B,A}}{\intersection{B,A}}
      = \symdiff{B}{A}$.
  \end{fproof}
