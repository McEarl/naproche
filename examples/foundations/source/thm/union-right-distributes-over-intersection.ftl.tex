  \begin{fproposition*}[label=3761728443645952]
    Let $A, B, C$ be classes.
    Then \[\union{\intersection{A,B},C} = \intersection{\union{A,C},\union{B,C}}.\]
  \end{fproposition*}
  \begin{fproof}
    $\union{\intersection{A,B},C}
      = \union{C,\intersection{A,B}}
      = \intersection{\union{C,A},\union{C,B}}
      = \intersection{\union{A,C},\union{B,C}}$.
  \end{fproof}
