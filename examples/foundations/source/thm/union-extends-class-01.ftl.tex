  \begin{fproposition*}[label=8894188219269120]
    Let $A, B$ be classes.
    Then $A \subclass[eq] \union{A,B}$.
  \end{fproposition*}
  \begin{fproof}
    Let $x \in A$.
    Then $x \in A$ or $x \in B$.
    Hence $x \in \union{A,B}$.
  \end{fproof}
