  \begin{fproposition*}[label=7664413741940736]
    Let $A$ be a class.
    Then \[\symdiff{A}{A} = \emptyclass.\]
  \end{fproposition*}
  \begin{fproof}
    $\symdiff{A}{A}
      = \complement{\union{A,A}}{\intersection{A,A}}
      = \complement{A}{A}
      = \emptyclass$.
  \end{fproof}
