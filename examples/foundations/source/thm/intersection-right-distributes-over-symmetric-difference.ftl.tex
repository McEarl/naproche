  \begin{fproposition*}[label=3761728443645952]
    Let $A, B, C$ be classes.
    Then \[\intersection{\symdiff{A}{B},C} = \symdiff{\intersection{A,C}}{\intersection{B,C}}.\]
  \end{fproposition*}
  \begin{fproof}
    $\intersection{\symdiff{A}{B},C}
      = \intersection{C,\symdiff{A}{B}}
      = \symdiff{\intersection{C,A}}{\intersection{C,B}}
      = \symdiff{\intersection{A,C}}{\intersection{B,C}}$.
  \end{fproof}
