  \begin{fproposition*}[label=6505712620404736]
    Let $A$ be a class.
    Then $A \union A = A$.
  \end{fproposition*}
  \begin{fproof}
    $A \union A = \compterm{x}{\text{$x \in A$ or $x \in A$}}$.
    Hence $A \union A = \compterm{x}{x \in A}$.
    Thus $A \union A = A$.
  \end{fproof}
