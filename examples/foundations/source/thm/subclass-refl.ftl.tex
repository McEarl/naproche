  \begin{proposition}[label=5994555614691328]
    Let $A$ be a class.
    $A \subseteq A$.
  \end{proposition}
  \begin{fproof*}
    Every element of $A$ is contained in $A$.
    Therefore $A \subseteq A$.
  \end{fproof*}
