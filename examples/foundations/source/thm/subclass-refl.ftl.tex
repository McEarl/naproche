  \begin{fproposition*}[label=5994555614691328]
    Let $A$ be a class.
    $A \subclass[eq] A$.
  \end{fproposition*}
  \begin{fproof}
    Every element of $A$ is contained in $A$.
    Therefore $A \subclass[eq] A$.
  \end{fproof}
