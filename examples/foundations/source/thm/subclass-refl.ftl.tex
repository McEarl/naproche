    \begin{proposition}\printlabel{5994555614691328}
      Let $A$ be a class.
      $A \subseteq A$.
    \end{proposition}
    \begin{proof}
      Every element of $A$ is contained in $A$.
      Therefore $A \subseteq A$.
    \end{proof}
