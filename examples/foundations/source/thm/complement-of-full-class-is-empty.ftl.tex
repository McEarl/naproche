  \begin{fproposition*}[label=7846433484963840]
    Let $A$ be a class.
    Then $A \complement A = \emptyclass$.
  \end{fproposition*}
  \begin{fproof}
    $A \complement A$ has no elements.
    Indeed $A \complement A = \fclass{x}{\text{$x \in A$ and $x \notin A$}}$.
    Hence the thesis.
  \end{fproof}
