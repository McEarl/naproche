  \begin{fproposition*}[label=8287827479494656]
    Let $A, B$ be classes.
    Then $\intersection{A,B} \subclass[eq] B$.
  \end{fproposition*}
  \begin{fproof}
    Let $x \in \intersection{A,B}$.
    Then $x \in A$ and $x \in B$.
    Hence $x \in B$.
  \end{fproof}
