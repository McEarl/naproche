  \begin{fproposition*}[label=2736947898351616]
    Let $A$ be a class.
    Then \[\symdiff{A}{\emptyclass} = A. \]
  \end{fproposition*}
  \begin{fproof}
    $\symdiff{A}{\emptyclass}
      = \complement{\union{A,\emptyclass}}{\intersection{A,\emptyclass}}
      = \complement{A}{\emptyclass}
      = A$.
  \end{fproof}
