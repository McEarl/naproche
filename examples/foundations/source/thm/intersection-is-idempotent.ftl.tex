  \begin{fproposition*}[label=5333724497444864]
    Let $A$ be a class.
    Then $\intersection{A,B} = A$.
  \end{fproposition*}
  \begin{fproof}
    $\intersection{A,B} = \fclass{x}{\text{$x \in A$ and $x \in A$}}$.
    Hence $\intersection{A,B} = \fclass{x}{x \in A}$.
    Thus $\intersection{A,B} = A$.
  \end{fproof}
