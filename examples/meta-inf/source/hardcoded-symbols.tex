\begin{smodule}[creators={Marcel Schütz}]{hardcoded-symbols}

  \symdecl*{class}
  \symdecl*{set}
  \symdecl*{map}
  \symdecl*{function}
  \symdecl*{object}
  \symdecl*{element}

  \symdef{equals}[args=2]{#1\mathrel{\comp=}#2}
  \symdef{nequals}[args=2]{#1\mathrel{\comp\neq}#2}
  \symdef{elem}[args=2]{#1\mathrel{\comp\in}#2}
  \symdef{nelem}[args=2]{#1\mathrel{\comp\notin}#2}
  \symdef{apply}[args=2]{#1\comp(#2\comp)}
  \symdef{domain}[args=1]{\comp{\textrm{dom}}\comp(#1\comp)}
  \symdef{tuple}[args=a]{\comp(\argsep{#1}{\comp,}\comp)}
  \symdef{set}[args=a]{\comp\{\argsep{#1}{\comp,}\comp\}}
  \symdef{compterm}[args=2]{\comp\{#1\;\comp\vert\;#2\comp\}}
  \symdef{iless}[args=2]{#1\mathrel{\maincomp\prec}#2}

  \begin{sdefinition}
    A \definame{class} is a notion.
  \end{sdefinition}

  \begin{sdefinition}
    A \definame{hardcoded-symbols?set} is a \sn{class} that is an \sn{hardcoded-symbols?object}.
  \end{sdefinition}

  \begin{sdefinition}
    A \definame{map} is a notion.
  \end{sdefinition}

  \begin{sdefinition}
    A \definame{function} is a \sn{map} that is an \sn{hardcoded-symbols?object}.
  \end{sdefinition}

  \begin{sdefinition}
    An \definame{hardcoded-symbols?object} is a notion.
    Let “a \definiendum{hardcoded-symbols?object}{mathematical object}” stand for “an \sn{hardcoded-symbols?object}”.
  \end{sdefinition}

  \begin{sdefinition}[for={equals,nequals}]
    \vardef{avar}{a}
    \vardef{bvar}{b}
    %
    Let $\avar,\bvar$ be arbitrary entities.
    $\defnotation{\equals{\avar}{\bvar}}$ iff there is no property that discerns $\avar$ and $\bvar$.
    Let “$\defnotation{\nequals{\avar}{\bvar}}$” stand for “not $\equals{\avar}{\bvar}$”.
  \end{sdefinition}

  \begin{sdefinition}[for={elem,nelem}]
    \vardef{Avar}{A}
    \vardef{avar}{a}
    %
    Let $\Avar$ be a \sn{class}.
    An \definame{element} of $\Avar$ is an \sn{hardcoded-symbols?object}.
    Let “$\defnotation{\elem{\avar}{\Avar}}$” stand for “$\avar$ is an \sn{element} of $\Avar$”.
    Let “$\defnotation{\nelem{\avar}{\Avar}}$” stand for “not $\elem{\avar}{\Avar}$”.
  \end{sdefinition}

  \begin{sdefinition}[for={apply}]
    \vardef{Fvar}{F}
    \vardef{avar}{a}
    %
    Let $\Fvar$ be a \sn{map} and $\elem{\avar}{\domain{\Fvar}}$.
    $\defnotation{\apply{\Fvar}{\avar}}$ is an \sn{hardcoded-symbols?object}.
  \end{sdefinition}

  \begin{sdefinition}[for={domain}]
    \vardef{Fvar}{F}
    %
    Let $\Fvar$ be a \sn{map}.
    $\defnotation{\domain{\Fvar}}$ is a \sn{class}.
  \end{sdefinition}

  \begin{sdefinition}[for={tuple}]
    \vardef{avar}{a}
    \vardef{bvar}{b}
    %
    Let $\avar,\bvar$ be \sns{hardcoded-symbols?object}.
    $\defnotation{\tuple{\avar,\bvar}}$ is an \sn{hardcoded-symbols?object}.
  \end{sdefinition}

  \begin{sdefinition}[for={set}]
    \vardef{nvar}{n}
    \varseq{aseq}{1,\ellipses,\nvar}{\maincomp{a}_{#1}}
    %
    Let $\aseq!$ be entities.
    $\defnotation{\set{\aseq{1},\ellipses,\aseq{\nvar}}}$ is the \sn{hardcoded-symbols?set} whose \sns{element} are precisely all \sns{hardcoded-symbols?object} among $\aseq!$.
  \end{sdefinition}

  \begin{sdefinition}[for={compterm}]
    \vardef{phivar}[args=1]{\maincomp{\varphi}\dobrackets{#1}}
    \vardef{xvar}{x}
    \vardef{avar}{a}
    \vardef{Xvar}{X}
    %
    Let $\phivar!$ be a formula and $\xvar$ a fresh variable.
    \begin{enumerate}
      \item $\compterm{\xvar}{\phivar!}$ is the \sn{class} whose \sns{element} are precisely all \sns{hardcoded-symbols?object} $\avar$ for which $\phivar{\avar}$ holds.
      \item Let, moreover, $\Xvar$ be a \sn{class}.
      $\compterm{\xvar \in \Xvar}{\phivar!}$ is the \sn{class} whose \sns{element} are precisely all \sns{element} of $\Xvar$ for which $\phivar{\avar}$ holds.
    \end{enumerate}
  \end{sdefinition}

  \begin{sdefinition}[for={iless}]
    \vardef{nvar}{n}
    \varseq{vseq}{1,\ellipses,\nvar}{\maincomp{v}_{#1}}
    \varseq{psiseq}[args=2]{1,\ellipses,\nvar}{\maincomp{\psi}_{#1}\dobrackets{#2}}
    \vardef{phivar}[args=1]{\maincomp{\varphi}\dobrackets{#1}}
    \vardef{tvar}[args=1]{\maincomp{t}\dobrackets{#1}}
    \varseq{xseq}{1,\ellipses,\nvar}{\maincomp{x}_{#1}}
    \varseq{yseq}{1,\ellipses,\nvar}{\maincomp{y}_{#1}}
    \vardef{xvar}{x}
    \vardef{yvar}{y}
    %
    $\defnotation{\iless!}$ is a binary relation between terms such that
    $\foral{\xvar}
        {\imply
          {\foral{\yvar}
            {\imply
              {\iless{\tvar{\yvar}}{\tvar{\xvar}}}
              {\phivar{\yvar}}
            }
          }
          {\phivar{\xvar}}}$
    implies $\foral{\xvar}{\phivar{\xvar}}$ for any formula $\phivar!$ and any term $\tvar!$.
  \end{sdefinition}
